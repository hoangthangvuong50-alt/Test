-- BananaTheme.lua
-- Chạy sau khi GUI/Window đã tồn tại. Script này "restyle" các phần tử GUI
-- Không thay đổi logic sự kiện, chỉ chỉnh giao diện (background, border, corner, text)

-- === CẤU HÌNH THEME (thay đổi ở đây nếu muốn) ===
local THEME = {
    MainBackground = Color3.fromRGB(18, 18, 20),       -- background tổng
    Panel = Color3.fromRGB(30, 30, 30),                -- panel chính của window
    Accent = Color3.fromRGB(255,196,37),               -- accent (vàng "banana")
    Accent2 = Color3.fromRGB(0,200,120),               -- accent phụ (xanh)
    Text = Color3.fromRGB(240,240,240),                -- màu chữ chính
    MutedText = Color3.fromRGB(170,170,170),           -- chữ phụ
    Border = Color3.fromRGB(60,60,60),                 -- border
    CornerRadius = UDim.new(0, 8),                     -- bo góc mặc định
    StrokeThickness = 1
}

-- === HÀM TIỆN ÍCH ===
local function applyUICorner(parent, corner)
    -- corner: UDim
    local existing = parent:FindFirstChildOfClass("UICorner")
    if existing then
        existing.CornerRadius = corner
    else
        local c = Instance.new("UICorner")
        c.CornerRadius = corner
        c.Parent = parent
    end
end

local function applyUIStroke(parent, color, thickness)
    local s = parent:FindFirstChildOfClass("UIStroke")
    if s then
        s.Color = color
        s.Thickness = thickness or 1
        s.LineJoinMode = Enum.LineJoinMode.Round
    else
        local st = Instance.new("UIStroke")
        st.Color = color
        st.Thickness = thickness or 1
        st.LineJoinMode = Enum.LineJoinMode.Round
        st.Parent = parent
    end
end

local function styleTextLabel(lbl, isTitle)
    if not lbl:IsA("TextLabel") and not lbl:IsA("TextButton") and not lbl:IsA("TextBox") then return end
    lbl.TextColor3 = isTitle and THEME.Text or THEME.MutedText
    lbl.Font = Enum.Font.GothamSemibold
    lbl.TextSize = isTitle and 16 or 14
    lbl.BackgroundTransparency = 1
end

local function styleButton(btn)
    if not btn:IsA("TextButton") then return end
    btn.BackgroundColor3 = THEME.Panel
    btn.TextColor3 = THEME.Text
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 14
    applyUICorner(btn, THEME.CornerRadius)
    applyUIStroke(btn, THEME.Border, THEME.StrokeThickness)
    -- hover effect (client-side)
    if btn:IsA("TextButton") then
        btn.MouseEnter:Connect(function()
            btn.BackgroundColor3 = THEME.Accent2
        end)
        btn.MouseLeave:Connect(function()
            btn.BackgroundColor3 = THEME.Panel
        end)
    end
end

local function styleFrame(frame)
    if not frame:IsA("Frame") and not frame:IsA("ScrollingFrame") and not frame:IsA("ImageLabel") then return end
    frame.BackgroundColor3 = THEME.Panel
    applyUICorner(frame, THEME.CornerRadius)
    applyUIStroke(frame, THEME.Border, THEME.StrokeThickness)
end

local function styleToggle(toggleFrame)
    -- Toggle cấu trúc tùy GUI của bạn, cố gắng áp dụng chung:
    -- toggleFrame contains a ToggleButton (a small frame) and a Label
    local toggleBtn = toggleFrame:FindFirstChild("ToggleButton") or toggleFrame:FindFirstChildWhichIsA("TextButton")
    if toggleBtn then
        toggleBtn.BackgroundColor3 = Color3.fromRGB(40,40,40)
        applyUICorner(toggleBtn, UDim.new(0, 12))
        applyUIStroke(toggleBtn, THEME.Border)
        if toggleBtn:IsA("TextButton") then
            toggleBtn.TextColor3 = THEME.Text
            toggleBtn.Font = Enum.Font.Gotham
            toggleBtn.TextSize = 12
        end
    end
    local lbl = toggleFrame:FindFirstChildWhichIsA("TextLabel")
    if lbl then styleTextLabel(lbl, false) end
end

local function styleSlider(sliderFrame)
    -- SliderFrame: tìm bar và thumb
    local bar = sliderFrame:FindFirstChild("Bar") or sliderFrame:FindFirstChildWhichIsA("Frame")
    if bar then
        bar.BackgroundColor3 = Color3.fromRGB(50,50,50)
        applyUICorner(bar, UDim.new(0, 6))
        applyUIStroke(bar, THEME.Border)
        local thumb = bar:FindFirstChild("Thumb") or sliderFrame:FindFirstChild("Thumb")
        if thumb then
            thumb.BackgroundColor3 = THEME.Accent
            applyUICorner(thumb, UDim.new(0, 8))
            applyUIStroke(thumb, THEME.Border, 1)
        end
    end
    local lbl = sliderFrame:FindFirstChildWhichIsA("TextLabel")
    if lbl then styleTextLabel(lbl, false) end
end

-- === HÀM CHÍNH: DUYỆT VÀ ÁP STYLE ===
local function restyleGui(root)
    if not root or not root:IsA("Instance") then return end

    -- nếu root là Window (Frame chứa header + body)
    if root:IsA("ScreenGui") then
    root.ResetOnSpawn = false
    local bg = root:FindFirstChild("BananaBackground")
    if not bg then
        bg = Instance.new("Frame")
        bg.Name = "BananaBackground"
        bg.Size = UDim2.new(1,0,1,0)
        bg.Position = UDim2.new(0,0,0,0)
        bg.BackgroundColor3 = THEME.MainBackground
        bg.BackgroundTransparency = 0.4 -- 👈 thêm dòng này để không bị che đen
        bg.BorderSizePixel = 0
        bg.ZIndex = -1 -- 👈 gửi nó ra sau tất cả phần tử UI
        bg.Parent = root
    else
        bg.BackgroundColor3 = THEME.MainBackground
        bg.ZIndex = -1
        bg.BackgroundTransparency = 0.4
    end
end

    for _, obj in pairs(root:GetDescendants()) do
        -- style frames / panels
        if obj:IsA("Frame") or obj:IsA("ScrollingFrame") then
            -- avoid changing background of root background we just added
            if obj.Name ~= "BananaBackground" then
                styleFrame(obj)
            end
        end

        -- style labels and titles
        if obj:IsA("TextLabel") then
            local isTitle = (obj.Name:lower():find("title") or obj.Name:lower():find("header") or obj.FontSize == Enum.FontSize.Size24)
            styleTextLabel(obj, isTitle)
        end

        -- style buttons
        if obj:IsA("TextButton") then
            styleButton(obj)
        end

        -- Style ImageButton similarly to buttons
        if obj:IsA("ImageButton") then
            obj.BackgroundColor3 = THEME.Panel
            applyUICorner(obj, THEME.CornerRadius)
            applyUIStroke(obj, THEME.Border)
        end

        -- Toggles (common naming)
        if obj.Name:lower():find("toggle") or obj.ClassName == "Frame" and obj.Name:lower():find("switch") then
            pcall(function() styleToggle(obj) end)
        end

        -- Sliders (common naming)
        if obj.Name:lower():find("slider") or obj.ClassName == "Frame" and obj.Name:lower():find("bar") then
            pcall(function() styleSlider(obj) end)
        end

        -- Dropdowns / Comboboxes: restyle container and options
        if obj.Name:lower():find("dropdown") or obj.Name:lower():find("comb") then
            if obj:IsA("Frame") then
                obj.BackgroundColor3 = THEME.Panel
                applyUICorner(obj, THEME.CornerRadius)
                applyUIStroke(obj, THEME.Border)
                for _, child in pairs(obj:GetChildren()) do
                    if child:IsA("TextLabel") then styleTextLabel(child, false) end
                    if child:IsA("TextButton") then styleButton(child) end
                end
            end
        end
    end
end

-- === SỬ DỤNG ===
-- Gọi restyleGui() với ScreenGui hoặc Frame gốc của window bạn muốn thay theme.
-- Ví dụ nếu Fluent tạo ScreenGui tên "FluentUI" thì:
local function applyToFluent()
    local player = game:GetService("Players").LocalPlayer
    if not player then return end
    local playerGui = player:WaitForChild("PlayerGui")

    -- THAY ĐỔI TÊN NÀY THEO GUI CỦA BẠN
    local targetGui = playerGui:FindFirstChild("FluentUI") or playerGui:FindFirstChildWhichIsA("ScreenGui")
    if not targetGui then
        warn("[BananaTheme] Không tìm thấy GUI mục tiêu (FluentUI). Thay tên trong script nếu cần.")
        return
    end

    restyleGui(targetGui)

    -- Nếu GUI của bạn tạo mục mới sau khi load (dynamic), bạn có thể kết nối:
    targetGui.DescendantAdded:Connect(function(desc)
        -- đợi 0 để đảm bảo properties đã set
        wait(0.03)
        restyleGui(targetGui)
    end)
end

-- delay cho GUI tạo xong
spawn(function()
    wait(0.5)
    applyToFluent()
end)
